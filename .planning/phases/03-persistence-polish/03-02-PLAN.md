```
const STORAGE_KEY = 'drp_scenarios'

export function isValidScenario(entry: unknown): entry is SavedScenario { ... }

export function loadFromStorage(): SavedScenario[] { ... }  // try/catch, sort desc, discard invalid

function persist(scenarios: SavedScenario[]): void { ... }  // try/catch, log on error

export function useScenarios() {
  const [scenarios, setScenarios] = useState<SavedScenario[]>(loadFromStorage)

  function save(name: string, data: WizardFormData): void { ... }
  function remove(id: string): void { ... }

  return { scenarios, save, remove }
}
```

Key implementation notes:
- isValidScenario: check typeof entry === 'object' && entry !== null, then cast and check id/name/date are strings, data is object, data.assets is object, Array.isArray(data.libraries), typeof data.san === 'object'
- loadFromStorage: JSON.parse in try/catch, check Array.isArray, filter with isValidScenario, sort by b.date.localeCompare(a.date) (ISO strings sort correctly as strings)
- save: prepend new entry [entry, ...scenarios] — already in newest-first order; persist; setScenarios
- persist: localStorage.setItem(STORAGE_KEY, JSON.stringify(scenarios)) in try/catch; on error console.error
- crypto.randomUUID() for id — available in all modern browsers, no import needed
- Export isValidScenario and loadFromStorage for unit testing (they are pure/side-effect-light functions)
